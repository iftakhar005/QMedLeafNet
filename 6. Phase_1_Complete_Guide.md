This is the **most critical foundation** of your entire project. Your job is to:

1. **Download and audit** the Bangladeshi medicinal plant dataset from Mendeley
2. **Create a master manifest** that tracks every file and its parent-image relationship
3. **Establish leakage-safe splits** using parent-image grouping (not file-level grouping)
4. **Lock down the label ontology** before any modeling begins

**Why this matters:** The dataset contains 1,981 original images and 20,019 augmented images. If augmented versions leak across train/test splits, your results will be **completely invalidated**. The grouped split approach prevents this entirely.

```python
#!/usr/bin/env python3
"""
Comprehensive dataset audit for the medicinal plant health detection project.
Verifies file counts, image integrity, and creates baseline statistics.
"""

import os
import json
import hashlib
from pathlib import Path
from collections import defaultdict
import numpy as np
from PIL import Image
from tqdm import tqdm
import pandas as pd

def compute_file_hash(filepath, algorithm='sha256'):
    """Compute hash of a file."""
    hash_obj = hashlib.new(algorithm)
    with open(filepath, 'rb') as f:
        for chunk in iter(lambda: f.read(4096), b''):
            hash_obj.update(chunk)
    return hash_obj.hexdigest()

def audit_directory_structure(raw_data_path):
    """Audit the directory structure."""
    audit_report = {}
    
    for folder in os.listdir(raw_data_path):
        folder_path = os.path.join(raw_data_path, folder)
        if os.path.isdir(folder_path):
            files = [f for f in os.listdir(folder_path) if os.path.isfile(os.path.join(folder_path, f))]
            audit_report[folder] = {
                'file_count': len(files),
                'sample_files': files[:5]
            }
    
    return audit_report

def check_image_integrity(raw_data_path, output_report_path):
    """Check all images for corruption and collect statistics."""
    
    image_stats = {
        'total_files': 0,
        'valid_images': 0,
        'corrupted_images': 0,
        'by_folder': defaultdict(lambda: {'valid': 0, 'corrupted': 0}),
        'size_histogram': defaultdict(int),
        'format_distribution': defaultdict(int),
        'corrupted_files': []
    }
    
    # Walk through all image files
    for root, dirs, files in os.walk(raw_data_path):
        for filename in tqdm(files, desc=f"Checking {os.path.basename(root)}"):
            if not filename.lower().endswith(('.jpg', '.jpeg', '.png', '.gif', '.bmp')):
                continue
            
            filepath = os.path.join(root, filename)
            image_stats['total_files'] += 1
            folder_name = os.path.basename(root)
            
            try:
                # Try to open and validate the image
                img = Image.open(filepath)
                img.verify()
                
                # Reopen after verify (verify closes the file)
                img = Image.open(filepath)
                width, height = img.size
                size_key = f"{width}x{height}"
                
                image_stats['valid_images'] += 1
                image_stats['by_folder'][folder_name]['valid'] += 1
                image_stats['size_histogram'][size_key] += 1
                image_stats['format_distribution'][img.format] += 1
                
            except Exception as e:
                image_stats['corrupted_images'] += 1
                image_stats['by_folder'][folder_name]['corrupted'] += 1
                image_stats['corrupted_files'].append({
                    'file': filepath,
                    'error': str(e)
                })
    
    return image_stats

def create_audit_report(raw_data_path, output_dir):
    """Create comprehensive audit report."""
    
    print("\n" + "="*80)
    print("DATASET AUDIT STARTING")
    print("="*80 + "\n")
    
    # Step 1: Directory structure
    print("Step 1: Auditing directory structure...")
    dir_audit = audit_directory_structure(raw_data_path)
    print(json.dumps(dir_audit, indent=2))
    
    # Step 2: Image integrity
    print("\nStep 2: Checking image integrity (this may take a few minutes)...")
    image_stats = check_image_integrity(raw_data_path, output_dir)
    
    # Step 3: Create report
    report = {
        'audit_date': pd.Timestamp.now().isoformat(),
        'raw_data_path': str(raw_data_path),
        'directory_structure': dir_audit,
        'image_statistics': {
            'total_files_checked': image_stats['total_files'],
            'valid_images': image_stats['valid_images'],
            'corrupted_images': image_stats['corrupted_images'],
            'corruption_rate': image_stats['corrupted_images'] / max(1, image_stats['total_files']),
            'size_distribution': dict(sorted(image_stats['size_histogram'].items())),
            'format_distribution': dict(image_stats['format_distribution']),
            'by_folder': dict(image_stats['by_folder']),
            'corrupted_files': image_stats['corrupted_files'][:10]  # First 10 for reference
        }
    }
    
    # Save report
    output_path = os.path.join(output_dir, 'data_audit.json')
    with open(output_path, 'w') as f:
        json.dump(report, f, indent=2)
    
    print("\n" + "="*80)
    print("AUDIT REPORT SUMMARY")
    print("="*80)
    print(f"Total files checked: {report['image_statistics']['total_files_checked']}")
    print(f"Valid images: {report['image_statistics']['valid_images']}")
    print(f"Corrupted images: {report['image_statistics']['corrupted_images']}")
    print(f"Corruption rate: {report['image_statistics']['corruption_rate']:.2%}")
    print(f"\nImage size distribution:")
    for size, count in sorted(report['image_statistics']['size_distribution'].items()):
        print(f"  {size}: {count} images")
    print(f"\nFormats found: {list(report['image_statistics']['format_distribution'].keys())}")
    print(f"\nFull report saved to: {output_path}")
    print("="*80 + "\n")
    
    return report

if __name__ == "__main__":
    raw_data_path = "data/raw"
    output_dir = "reports"
    
    os.makedirs(output_dir, exist_ok=True)
    report = create_audit_report(raw_data_path, output_dir)
```

```python
#!/usr/bin/env python3
"""
Extract metadata from filenames and any provided metadata CSV.
Build the definitive label ontology for the project.
"""

import os
import re
import pandas as pd
from pathlib import Path
from collections import defaultdict
from tqdm import tqdm

class MetadataExtractor:
    """Extract metadata from the medicinal plant dataset."""
    
    def __init__(self, raw_data_path):
        self.raw_data_path = raw_data_path
        self.filename_patterns = self._build_patterns()
        self.metadata = []
        
    def _build_patterns(self):
        """
        Build regex patterns for filename parsing.
        Adjust these based on actual filename format from the dataset.
        
        Expected formats might be:
        - [species]_[condition]_[id]_[augmentation].jpg
        - [species]_leaf_[condition]_[number].jpg
        - etc.
        """
        # This is a template - adjust based on actual filenames
        return {
            'species_condition': re.compile(
                r'(.+?)_(.+?)_(\d+)(?:_aug)?(?:\d+)?\.jpg',
                re.IGNORECASE
            ),
        }
    
    def extract_from_filenames(self):
        """Parse filenames to extract species, condition, and identifiers."""
        
        metadata_records = []
        
        for root, dirs, files in os.walk(self.raw_data_path):
            folder_name = os.path.basename(root)
            is_original = 'Augmented' not in folder_name
            is_background_removed = 'Background_Removed' in folder_name
            
            for filename in tqdm(files, desc=f"Parsing {folder_name}"):
                if not filename.lower().endswith(('.jpg', '.jpeg', '.png')):
                    continue
                
                filepath = os.path.join(root, filename)
                relative_path = os.path.relpath(filepath, self.raw_data_path)
                
                record = {
                    'filename': filename,
                    'filepath': filepath,
                    'relative_path': relative_path,
                    'folder': folder_name,
                    'is_original': is_original,
                    'is_background_removed': is_background_removed,
                    'species': None,
                    'condition': None,
                    'parent_id': None,
                }
                
                # Try to parse filename (adjust pattern as needed)
                # For now, we'll document the structure
                record['raw_filename'] = filename
                
                metadata_records.append(record)
        
        return pd.DataFrame(metadata_records)
    
    def load_provided_metadata(self, metadata_csv_path):
        """Load metadata CSV if provided by dataset authors."""
        if os.path.exists(metadata_csv_path):
            df = pd.read_csv(metadata_csv_path)
            print(f"Loaded metadata CSV: {len(df)} rows")
            print(f"Columns: {df.columns.tolist()}")
            return df
        return None
    
    def build_label_ontology(self, df, provided_metadata=None):
        """
        Build the definitive label ontology.
        This is crucial - lock it down before any modeling.
        """
        
        ontology = {
            'species': [],
            'conditions': [],
            'species_condition_combinations': {}
        }
        
        if provided_metadata is not None:
            if 'species' in provided_metadata.columns:
                ontology['species'] = sorted(provided_metadata['species'].unique().tolist())
            if 'condition' in provided_metadata.columns:
                ontology['conditions'] = sorted(provided_metadata['condition'].unique().tolist())
        
        # If we parsed it from filenames, validate
        if 'species' in df.columns and df['species'].notna().any():
            parsed_species = sorted(df['species'].dropna().unique().tolist())
            if not ontology['species']:
                ontology['species'] = parsed_species
            print(f"Unique species found: {len(ontology['species'])}")
            print(f"Species list: {ontology['species']}")
        
        if 'condition' in df.columns and df['condition'].notna().any():
            parsed_conditions = sorted(df['condition'].dropna().unique().tolist())
            if not ontology['conditions']:
                ontology['conditions'] = parsed_conditions
            print(f"Unique conditions found: {len(ontology['conditions'])}")
            print(f"Condition list: {ontology['conditions']}")
        
        return ontology
```

## STEP 5: Build Master Manifest with Parent-Image Grouping

### 5.1 Create Manifest Builder Script

Create `src/tools/build_manifest.py`:

```python
#!/usr/bin/env python3
"""
Build the master manifest with parent-image grouping.
This is the foundation of leakage-safe evaluation.

Key principle: Every augmented image must map to exactly one original parent.
"""

import os
import pandas as pd
import hashlib
import re
from pathlib import Path
from PIL import Image
from tqdm import tqdm
from collections import defaultdict
import json

class ManifestBuilder:
    """Build leakage-safe manifest with parent-image tracking."""
    
    def __init__(self, raw_data_path, label_ontology_path):
        self.raw_data_path = raw_data_path
        self.records = []
        
        # Load label ontology
        with open(label_ontology_path) as f:
            self.ontology = json.load(f)
    
    def extract_parent_id(self, filename, folder_name):
        """
        Extract parent image ID from filename.
        
        The parent_id should be:
        - For original images: the full ID
        - For augmented images: the ID of the original it was derived from
        
        Adjust the regex based on actual filename patterns.
        """
        
        # Template patterns - adjust based on actual filenames
        # Example: leaf_001_healthy.jpg → parent_id = "leaf_001"
        # Example: leaf_001_healthy_aug_0.jpg → parent_id = "leaf_001"
        
        if 'Augmented' in folder_name:
            # Remove augmentation suffix
            parent = re.sub(r'_aug(mented)?.*$', '', filename, flags=re.IGNORECASE)
            parent = re.sub(r'_\d+\.(jpg|jpeg|png)$', f'.\\1', parent, flags=re.IGNORECASE)
        else:
            parent = filename
        
        return parent
    
    def build_original_index(self):
        """
        Create an index of all original images.
        Every augmented image must match an original.
        """
        originals = {}
        
        for root, dirs, files in os.walk(self.raw_data_path):
            if 'Original' in root and 'Augmented' not in root:
                for filename in files:
                    if filename.lower().endswith(('.jpg', '.jpeg', '.png')):
                        parent_id = self.extract_parent_id(filename, root)
                        originals[parent_id] = {
                            'filename': filename,
                            'folder': os.path.basename(root),
                            'path': os.path.join(root, filename)
                        }
        
        return originals
    
    def build_manifest(self, ontology_file='reports/label_ontology.json'):
        """Build complete manifest with parent-image tracking."""
        
        print("Building original image index...")
        original_index = self.build_original_index()
        print(f"Found {len(original_index)} original images")
        
        print("\nScanning all images and building manifest...")
        
        for root, dirs, files in os.walk(self.raw_data_path):
            folder_name = os.path.basename(root)
            is_original = 'Augmented' not in folder_name
            is_background_removed = 'Background_Removed' in folder_name
            
            for filename in tqdm(files, desc=folder_name):
                if not filename.lower().endswith(('.jpg', '.jpeg', '.png')):
                    continue
                
                filepath = os.path.join(root, filename)
                relative_path = os.path.relpath(filepath, self.raw_data_path)
                
                # Extract parent ID
                parent_id = self.extract_parent_id(filename, folder_name)
                
                # Verify parent exists (for augmented images)
                parent_found = parent_id in original_index or is_original
                
                record = {
                    'filename': filename,
                    'relative_path': relative_path,
                    'folder': folder_name,
                    'is_original': is_original,
                    'is_background_removed': is_background_removed,
                    'parent_leaf_id': parent_id,
                    'parent_found': parent_found,
                    'species': None,  # To be filled from ontology
                    'condition': None,  # To be filled from ontology
                    'file_size_bytes': os.path.getsize(filepath),
                }
                
                # Try to extract species and condition from filename
                # (adjust based on actual naming pattern)
                parts = filename.lower().split('_')
                
                # Match species
                for sp in self.ontology['species']:
                    if sp.lower() in filename.lower():
                        record['species'] = sp
                        break
                
                # Match condition
                for cond in self.ontology['conditions']:
                    if cond.lower() in filename.lower():
                        record['condition'] = cond
                        break
                
                self.records.append(record)
        
        return pd.DataFrame(self.records)
    
    def validate_manifest(self, df):
        """Perform integrity checks on the manifest."""
        
        errors = []
        warnings = []
        
        # Check 1: All parent IDs should map to an original
        unmapped_parents = df[~df['parent_found']]['parent_leaf_id'].unique()
        if len(unmapped_parents) > 0:
            errors.append(f"Found {len(unmapped_parents)} unmapped parent IDs: {unmapped_parents[:5].tolist()}")
        else:
            print("✓ All parent-image mappings are valid")
        
        # Check 2: Original images should be unique
        originals = df[df['is_original']]
        duplicate_originals = originals[originals.duplicated(subset=['parent_leaf_id'], keep=False)]
        if len(duplicate_originals) > 0:
            errors.append(f"Found {len(duplicate_originals)} duplicate original images")
        else:
            print("✓ All original images are unique")
        
        # Check 3: Check label coverage
        missing_species = df[df['species'].isna()]
        missing_condition = df[df['condition'].isna()]
        
        if len(missing_species) > 0:
            warnings.append(f"Could not extract species for {len(missing_species)} files")
        if len(missing_condition) > 0:
            warnings.append(f"Could not extract condition for {len(missing_condition)} files")
        
        print(f"\n✓ Species coverage: {(1 - len(missing_species)/len(df)):.1%}")
        print(f"✓ Condition coverage: {(1 - len(missing_condition)/len(df)):.1%}")
        
        # Check 4: Class distribution
        print(f"\nSpecies distribution:")
        print(df['species'].value_counts())
        print(f"\nCondition distribution:")
        print(df['condition'].value_counts())
        
        return errors, warnings
    
    def save_manifest(self, df, output_path='manifests/master_manifest.parquet'):
        """Save manifest in parquet format (compressed, preserves types)."""
        
        os.makedirs(os.path.dirname(output_path), exist_ok=True)
        df.to_parquet(output_path, index=False)
        print(f"\nMaster manifest saved to: {output_path}")
        print(f"Total records: {len(df)}")
        print(f"Columns: {df.columns.tolist()}")
        
        # Also save as CSV for inspection
        csv_path = output_path.replace('.parquet', '.csv')
        df.to_csv(csv_path, index=False)
        print(f"CSV version saved to: {csv_path}")
```

## STEP 6: Create Manifest Integrity Checker

### 6.1 Build Integrity Check Script

Create `src/tools/check_manifest.py`:

```python
#!/usr/bin/env python3
"""
Comprehensive integrity checks for the master manifest.
Run this before creating splits.
"""

import pandas as pd
import numpy as np
import json
from pathlib import Path

class ManifestChecker:
    """Verify manifest integrity and data quality."""
    
    def __init__(self, manifest_path):
        self.df = pd.read_parquet(manifest_path)
        self.errors = []
        self.warnings = []
        self.report = {}
    
    def check_parent_mapping(self):
        """Verify all parents map correctly."""
        print("Checking parent-image mappings...")
        
        # Check 1: All parent IDs should be mappable
        unmapped = self.df[~self.df['parent_found']]
        if len(unmapped) > 0:
            self.errors.append(f"{len(unmapped)} files have unmapped parents")
        else:
            print("  ✓ All parents mapped correctly")
        
        # Check 2: Each original should have consistent augmentations
        for parent_id in self.df['parent_leaf_id'].unique():
            group = self.df[self.df['parent_leaf_id'] == parent_id]
            species = group['species'].dropna().unique()
            condition = group['condition'].dropna().unique()
            
            if len(species) > 1:
                self.errors.append(f"Parent {parent_id} has inconsistent species: {species}")
            if len(condition) > 1:
                self.errors.append(f"Parent {parent_id} has inconsistent conditions: {condition}")
        
        if not self.errors:
            print("  ✓ All augmented images inherit correct labels from parents")
    
    def check_label_coverage(self):
        """Check label completeness."""
        print("Checking label coverage...")
        
        missing_species = self.df['species'].isna().sum()
        missing_condition = self.df['condition'].isna().sum()
        
        if missing_species > 0:
            self.warnings.append(f"{missing_species} files missing species labels ({missing_species/len(self.df):.1%})")
        if missing_condition > 0:
            self.warnings.append(f"{missing_condition} files missing condition labels ({missing_condition/len(self.df):.1%})")
        
        if missing_species == 0 and missing_condition == 0:
            print("  ✓ 100% label coverage")
    
    def check_class_balance(self):
        """Report class distribution."""
        print("Checking class distribution (at parent level)...")
        
        # Use only original images for true class distribution
        originals = self.df[self.df['is_original']].copy()
        
        species_dist = originals['species'].value_counts()
        condition_dist = originals['condition'].value_counts()
        
        self.report['species_distribution'] = species_dist.to_dict()
        self.report['condition_distribution'] = condition_dist.to_dict()
        
        # Check for extremely imbalanced classes
        if species_dist.min() / species_dist.max() < 0.1:
            self.warnings.append(f"Species classes highly imbalanced (ratio < 0.1)")
        
        if condition_dist.min() / condition_dist.max() < 0.1:
            self.warnings.append(f"Condition classes highly imbalanced (ratio < 0.1)")
        
        print(f"\n  Species distribution (among {len(originals)} originals):")
        for sp, count in species_dist.items():
            print(f"    {sp}: {count} ({count/len(originals)*100:.1f}%)")
        
        print(f"\n  Condition distribution:")
        for cond, count in condition_dist.items():
            print(f"    {cond}: {count} ({count/len(originals)*100:.1f}%)")
    
    def check_file_integrity(self):
        """Check that all files exist and have reasonable sizes."""
        print("Checking file integrity...")
        
        missing_files = []
        suspicious_sizes = []
        
        for idx, row in self.df.iterrows():
            filepath = Path('data/raw') / row['relative_path']
            
            if not filepath.exists():
                missing_files.append(row['filename'])
            
            # Check file size (very small or very large files)
            size_kb = row['file_size_bytes'] / 1024
            if size_kb < 10 or size_kb > 5000:  # Adjust based on typical sizes
                suspicious_sizes.append((row['filename'], size_kb))
        
        if missing_files:
            self.errors.append(f"{len(missing_files)} files missing from disk")
        else:
            print("  ✓ All manifest files exist on disk")
        
        if suspicious_sizes:
            if len(suspicious_sizes) > 0.01 * len(self.df):  # More than 1% suspicious
                self.warnings.append(f"Found {len(suspicious_sizes)} files with unusual sizes")
        else:
            print("  ✓ All file sizes look reasonable")
    
    def run_all_checks(self):
        """Run complete integrity check suite."""
        self.check_parent_mapping()
        self.check_label_coverage()
        self.check_class_balance()
        self.check_file_integrity()
    
    def print_report(self):
        """Print formatted check report."""
        print("\n" + "="*80)
        print("MANIFEST INTEGRITY CHECK REPORT")
        print("="*80 + "\n")
        
        print(f"Total files in manifest: {len(self.df)}")
        print(f"Original images: {self.df['is_original'].sum()}")
        print(f"Augmented images: {(~self.df['is_original']).sum()}")
        
        if self.errors:
            print(f"\n❌ CRITICAL ERRORS ({len(self.errors)}):")
            for err in self.errors:
                print(f"  - {err}")
        else:
            print(f"\n✓ No critical errors found")
        
        if self.warnings:
            print(f"\n⚠ WARNINGS ({len(self.warnings)}):")
            for warn in self.warnings:
                print(f"  - {warn}")
        else:
            print(f"\n✓ No warnings")
        
        print("\n" + "="*80)
        
        return len(self.errors) == 0

def main():
    checker = ManifestChecker('manifests/master_manifest.parquet')
    checker.run_all_checks()
    success = checker.print_report()
    
    if success:
        print("✓ MANIFEST INTEGRITY VERIFIED - Ready for split creation")
    else:
        print("❌ Fix errors before proceeding to next step")
    
    return success

if __name__ == "__main__":
    success = main()
    exit(0 if success else 1)
```

## STEP 7: Create Leakage-Safe Grouped Splits

### 7.1 Build Split Generator Script

Create `src/tools/make_grouped_splits.py`:

```python
#!/usr/bin/env python3
"""
Create leakage-safe, grouped, stratified train/val/test splits.

CRITICAL: This uses parent_leaf_id to ensure augmented siblings never cross boundaries.
"""

import pandas as pd
import numpy as np
from sklearn.model_selection import StratifiedKFold, train_test_split
import os
import json

class GroupedSplitGenerator:
    """Generate leakage-safe grouped splits."""
    
    def __init__(self, manifest_path, random_seed=3407):
        self.df = pd.read_parquet(manifest_path)
        self.seed = random_seed
        np.random.seed(random_seed)
        
        # Work with originals only
        self.originals = self.df[self.df['is_original']].copy()
        print(f"Working with {len(self.originals)} original images")
    
    def create_grouped_test_set(self, test_ratio=0.15):
        """
        Split at parent-image level, not file level.
        This ensures no data leakage.
        """
        print(f"\nCreating grouped test set ({test_ratio:.0%})...")
        
        unique_parents = self.originals['parent_leaf_id'].unique()
        print(f"Unique parent images: {len(unique_parents)}")
        
        # Get one representative record per parent
        parent_samples = self.originals.drop_duplicates(subset=['parent_leaf_id']).copy()
        
        # Stratified split at parent level
        test_parents, dev_parents = train_test_split(
            parent_samples,
            test_size=test_ratio,
            random_state=self.seed,
            stratify=parent_samples['condition']  # Or 'species' if too imbalanced
        )
        
        test_parent_ids = set(test_parents['parent_leaf_id'].values)
        dev_parent_ids = set(dev_parents['parent_leaf_id'].values)
        
        # Create test set (all files from test parents)
        test_set = self.df[self.df['parent_leaf_id'].isin(test_parent_ids)].copy()
        dev_pool = self.df[self.df['parent_leaf_id'].isin(dev_parent_ids)].copy()
        
        print(f"  Test parent groups: {len(test_parent_ids)}")
        print(f"  Dev parent groups: {len(dev_parent_ids)}")
        print(f"  Test files: {len(test_set)}")
        print(f"  Dev files: {len(dev_pool)}")
        
        return test_set, dev_pool, test_parent_ids, dev_parent_ids
    
    def create_grouped_cv_folds(self, dev_pool, n_folds=5):
        """
        Create grouped k-fold cross-validation at parent level.
        """
        print(f"\nCreating grouped {n_folds}-fold cross-validation...")
        
        # Get unique parents in dev pool
        dev_parents = dev_pool.drop_duplicates(subset=['parent_leaf_id']).copy()
        
        # Stratified k-fold at parent level
        skf = StratifiedKFold(n_splits=n_folds, shuffle=True, random_state=self.seed)
        
        folds = []
        for fold_idx, (train_indices, val_indices) in enumerate(skf.split(dev_parents, dev_parents['condition'])):
            train_parents = dev_parents.iloc[train_indices]['parent_leaf_id'].values
            val_parents = dev_parents.iloc[val_indices]['parent_leaf_id'].values
            
            # Get all files (including augmentations) for each group
            fold_train = dev_pool[dev_pool['parent_leaf_id'].isin(train_parents)].copy()
            fold_val = dev_pool[dev_pool['parent_leaf_id'].isin(val_parents)].copy()
            
            folds.append({
                'fold': fold_idx,
                'train': fold_train,
                'val': fold_val,
                'train_parent_ids': set(train_parents),
                'val_parent_ids': set(val_parents)
            })
            
            print(f"  Fold {fold_idx}: train={len(fold_train)} files ({len(train_parents)} parents), "
                  f"val={len(fold_val)} files ({len(val_parents)} parents)")
        
        return folds
    
    def verify_no_leakage(self, test_set, dev_pool, folds):
        """Verify there is no parent-image leakage across splits."""
        print("\n" + "="*80)
        print("VERIFYING NO DATA LEAKAGE")
        print("="*80)
        
        test_parents = set(test_set['parent_leaf_id'].values)
        
        # Check test vs dev
        overlap = test_parents & set(dev_pool['parent_leaf_id'].values)
        if overlap:
            print(f"❌ ERROR: {len(overlap)} parents in both test and dev!")
            return False
        else:
            print("✓ Test and dev sets have no parent overlap")
        
        # Check each fold
        for fold in folds:
            train_parents = fold['train_parent_ids']
            val_parents = fold['val_parent_ids']
            
            overlap = train_parents & val_parents
            if overlap:
                print(f"❌ ERROR in fold {fold['fold']}: {len(overlap)} parents in both train and val!")
                return False
            
            # Also check against test
            overlap_test = train_parents & test_parents
            if overlap_test:
                print(f"❌ ERROR in fold {fold['fold']}: {len(overlap_test)} parents in both train and test!")
                return False
            
            overlap_test_val = val_parents & test_parents
            if overlap_test_val:
                print(f"❌ ERROR in fold {fold['fold']}: {len(overlap_test_val)} parents in both val and test!")
                return False
        
        print(f"✓ All {len(folds)} folds verified - zero parent leakage")
        return True
    
    def save_splits(self, test_set, folds, output_dir='manifests/splits_v1'):
        """Save all splits to CSV files."""
        os.makedirs(output_dir, exist_ok=True)
        
        print(f"\nSaving splits to {output_dir}...")
        
        # Save test set
        test_output = os.path.join(output_dir, 'test_groups.csv')
        test_set.to_csv(test_output, index=False)
        print(f"  Saved: {test_output} ({len(test_set)} files)")
        
        # Save each fold
        for fold in folds:
            fold_idx = fold['fold']
            
            train_output = os.path.join(output_dir, f'dev_fold_{fold_idx}_train.csv')
            val_output = os.path.join(output_dir, f'dev_fold_{fold_idx}_val.csv')
            
            fold['train'].to_csv(train_output, index=False)
            fold['val'].to_csv(val_output, index=False)
            
            print(f"  Saved fold {fold_idx}: train={len(fold['train'])}, val={len(fold['val'])}")
        
        # Create split report
        report = {
            'random_seed': self.seed,
            'test_ratio': len(test_set) / len(self.df),
            'n_folds': len(folds),
            'total_files': len(self.df),
            'test_files': len(test_set),
            'dev_files': len(pd.concat([f['train'] for f in folds] + [f['val'] for f in folds]).drop_duplicates(subset=['filename'])),
            'test_parent_groups': len(set(test_set['parent_leaf_id'].values)),
            'verification': 'zero_parent_leakage'
        }
        
        report_output = os.path.join(output_dir, 'split_report.json')
        with open(report_output, 'w') as f:
            json.dump(report, f, indent=2)
        
        print(f"  Saved: {report_output}")
        
        return output_dir

def main():
    print("\n" + "="*80)
    print("CREATING LEAKAGE-SAFE GROUPED SPLITS")
    print("="*80)
    
    generator = GroupedSplitGenerator(
        manifest_path='manifests/master_manifest.parquet',
        random_seed=3407
    )
    
    # Create test set and dev pool
    test_set, dev_pool, test_parents, dev_parents = generator.create_grouped_test_set(test_ratio=0.15)
    
    # Create folds
    folds = generator.create_grouped_cv_folds(dev_pool, n_folds=5)
    
    # Verify no leakage
    if not generator.verify_no_leakage(test_set, dev_pool, folds):
        print("\n❌ LEAKAGE DETECTED - ABORTING")
        return False
    
    # Save splits
    generator.save_splits(test_set, folds)
    
    print("\n" + "="*80)
    print("GROUPED SPLITS CREATED SUCCESSFULLY")
    print("="*80 + "\n")
    
    return True

if __name__ == "__main__":
    success = main()
    exit(0 if success else 1)
```
