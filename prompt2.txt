QMEDLEAFNET PROJECT PIPELINE - HTML/WEB APPLICATION BUILD PROMPT
================================================================================

PROJECT OVERVIEW FOR WEB BUILD
================================================================================

PROJECT NAME: QMedLeafNet Web Dashboard
PURPOSE: Build an interactive web interface to manage, visualize, and execute the 
         medicinal plant leaf health detection pipeline
TARGET USERS: Data scientists, researchers, model trainers
TECHNOLOGY STACK: HTML5, CSS3, JavaScript (ES6+), fetch API for backend communication
DEPLOYMENT: Single-page application (SPA) or multi-page app served from Python backend

CORE FUNCTIONALITY:
1. Display project status and overview
2. Manage and visualize datasets
3. Configure and execute training jobs
4. Monitor feature extraction progress
5. View and compare model results
6. Display classification results and metrics
7. Visualize performance across different heads


================================================================================
SECTION 1: HIGH-LEVEL APPLICATION ARCHITECTURE
================================================================================

1.1 APPLICATION STRUCTURE
  The web application consists of:
    - Backend: Python (Flask/FastAPI) serving HTML + JSON APIs
    - Frontend: HTML/CSS/JavaScript for user interface
    - Data layer: CSV manifests and JSON result files
    - Visualization: Charts, tables, progress indicators

1.2 APPLICATION FLOW
  User opens browser
    ↓
  Flask server serves index.html
    ↓
  JavaScript loads and renders page
    ↓
  User interacts with UI (buttons, forms, etc.)
    ↓
  JavaScript sends fetch requests to Python backend
    ↓
  Python processes request (training, prediction, file I/O)
    ↓
  Backend returns JSON response
    ↓
  JavaScript updates DOM with results
    ↓
  User sees updated interface in real-time

1.3 PAGE STRUCTURE
  Main pages in application:
    - Dashboard (home page with overview)
    - Dataset Manager (upload, view, split data)
    - Model Configuration (select heads, set parameters)
    - Training Monitor (track training progress)
    - Results Viewer (view metrics and predictions)
    - Settings & Configuration
    - About/Documentation

  // Handle backbone selection
  document.querySelectorAll('input[name="backbone"]').forEach(radio => {
    radio.addEventListener('change', (e) => {
      const params = e.target.dataset.params
      document.getElementById('backbone-params').textContent = params
      updateSummary()
    })
  })
  
  // Handle head selection
  document.querySelectorAll('.head-card').forEach(card => {
    card.addEventListener('click', (e) => {
      if (e.target.tagName !== 'INPUT') {
        card.querySelector('input[type="checkbox"]').click()
      }
      card.classList.toggle('selected')
      updateSummary()
    })
  })
  
  // Update configuration summary
  function updateSummary() {
    const backbone = document.querySelector('input[name="backbone"]:checked').value
    const heads = Array.from(document.querySelectorAll('.head-card input:checked'))
      .map(cb => cb.value)
    const split = document.querySelector('input[name="split"]:checked').value
    
    const summary = `
      Backbone: ${backbone}
      Heads: ${heads.join(', ')}
      Split: ${split}
      Estimated time: [calculated]
    `
    document.getElementById('summary-box').textContent = summary
  }
  
  // Handle next button
  document.getElementById('next-button').addEventListener('click', () => {
    const config = getConfiguration()
    if (!validateConfiguration(config)) {
      alert('Please select at least one head')
      return
    }
    saveConfigurationAndNavigate('training-monitor', config)
  })

2.4.3 TRAINING MONITOR JAVASCRIPT
  // WebSocket or polling for real-time updates
  let trainingActive = false
  let updateInterval = null
  
  async function startTraining() {
    trainingActive = true
    const config = getStoredConfiguration()
    
    const response = await fetch('/api/start-training', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(config)
    })
    
    if (response.ok) {
      updateInterval = setInterval(updateTrainingStatus, 2000)
      document.getElementById('training-header').textContent = 'TRAINING IN PROGRESS'
    }
  }
  
  async function updateTrainingStatus() {
    const status = await fetch('/api/training-status').then(r => r.json())
    
    // Update progress bars
    document.getElementById('feature-extraction-progress').style.width = 
      status.feature_extraction_percent + '%'
    
    status.head_progress.forEach((progress, index) => {
      document.getElementById(`head-${index}-progress`).style.width = 
        progress.percent + '%'
    })
    
    // Update stats
    document.getElementById('images-processed').textContent = 
      status.images_processed + '/' + status.total_images
    
    // Append new logs
    appendLogs(status.new_logs)
    
    // Check if completed
    if (status.status === 'COMPLETED') {
      trainingActive = false
      clearInterval(updateInterval)
      showNotification('Training completed successfully!', 'success')
      navigateTo('results')
    }
    
    if (status.status === 'ERROR') {
      trainingActive = false
      clearInterval(updateInterval)
      showNotification('Training error: ' + status.error_message, 'error')
    }
  }
  
  function appendLogs(newLogs) {
    const logContainer = document.getElementById('process-log')
    newLogs.forEach(log => {
      const line = document.createElement('div')
      line.className = 'process-log-line'
      line.innerHTML = `
        <span class="log-timestamp">[${log.timestamp}]</span>
        <span class="log-message">${log.message}</span>
      `
      logContainer.appendChild(line)
    })
    logContainer.scrollTop = logContainer.scrollHeight
  }
  
  // Stop training button
  document.getElementById('stop-button').addEventListener('click', async () => {
    if (confirm('Are you sure you want to stop training?')) {
      await fetch('/api/stop-training', { method: 'POST' })
      trainingActive = false
      clearInterval(updateInterval)
    }
  })

  // Base API URL
  const API_BASE_URL = '/api'
  
  // Generic fetch wrapper with error handling
  async function apiCall(endpoint, options = {}) {
    const url = `${API_BASE_URL}${endpoint}`
    const response = await fetch(url, {
      headers: {
        'Content-Type': 'application/json',
        ...options.headers
      },
      ...options
    })
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status} ${response.statusText}`)
    }
    
    return await response.json()
  }
  
  // API endpoints
  const api = {
    // Project endpoints
    getProjectStatus: () => apiCall('/project-status'),
    getUser: () => apiCall('/user'),
    
    // Dataset endpoints

    uploadManifest: (file) => {
      const formData = new FormData()
      formData.append('file', file)
      return apiCall('/upload-manifest', {
        method: 'POST',
        headers: {},
        body: formData
      })
    },
    
    // Training endpoints
    startTraining: (config) => apiCall('/training/start', {
      method: 'POST',
      body: JSON.stringify(config)
